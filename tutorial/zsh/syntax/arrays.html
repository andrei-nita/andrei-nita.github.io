Arrays

Arrays store multiple values.

Declare an array

name="zsh"
# Declare an array with 6 elements
array0=(name two three four five six)

Print first element

echo $array0 # => "zsh"
# or
echo ${array0[0]} # => "zsh"

Save element to variable

firstValue=$array0
echo ${firstValue} # => "zsh"
# or
firstValue=${array0[0]}
echo ${firstValue} # => "zsh"

Print all elements

echo ${array0[@]} # => "one two three four five six"
# or
for i in "${array0[@]}"; do
    echo "$i"
done

Print number of elements

echo ${#array0[@]} # => "6"

Print number of characters in third element

echo ${#array0[2]} # => "5"

Print range of elements

# Print 2 elements starting from forth
echo ${array0[@]:3:2} # => "four five"
